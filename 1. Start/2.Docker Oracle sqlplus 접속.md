# 2.Docker Oracle sqlplus 접속
## docker sqlplus 접속
```
docker exec -it oracle sqlplus
user-name : system
password : 1234

# db의 모든 테이블 조회
SQL> select * from tab;

# oracle db 에 접속시 container database(CDB) 에 접속하게 된다
# 초기 이름은 ROOT 이다.
SQL> SHOW con_name;

# 접속할 수 있는(pluggable) database 를 생성해야한다.
# syntax) create pluggable database "PDB 이름" admin user "PDB 관리자 이름" identified by "PDB 관리자 패스워드" roles = (role) file_name_convert = ('소스 PDB 위치','PDB 생성 위치')
SQL> create pluggable database "TEST" admin user "USER" identified by "1234" roles=(dba) file_name_convert = ('/opt/oracle/oradata/ORCL/pdbseed','/opt/oracle/oradata/USER/HRPDB/') ;

```
